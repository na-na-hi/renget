#include <iostream>

using namespace std;

int main()
{
  //---
  cout << '\n';

  cout << 1234 << "\t(decimal)\n"
       << hex << 1234 << "\t(hexadecimal)\n"
       << oct << 1234 << "\t(octal)\n";

  //---
  cout << '\n';

  cout << dec << 1234 << '\t' << hex << 1234 << '\t' << oct << 1234 << '\n';
  cout << 1234 << '\n';  // the octal base is still in effect

  //---
  cout << '\n';

  cout << dec << 1234 << '\t' << hex << 1234 << '\t' << oct << 1234 << '\n';
  cout << showbase;  // show bases
  cout << dec << 1234 << '\t' << hex << 1234 << '\t' << oct << 1234 << '\n';

  //---
  cout << '\n';

  cout << dec << 1234 << '\t' << 0x4d2 << '\t' << 02322 << '\n';
}
